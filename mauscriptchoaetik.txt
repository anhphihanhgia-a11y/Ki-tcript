-- Định nghĩa các dịch vụ cần thiết
local Players = game:GetService("Players")
local CoreGui = game:GetService("CoreGui") -- Dùng cho Studio, nếu chạy thông thường thì dùng LocalPlayer.PlayerGui

local player = Players.LocalPlayer
local playerGui = player:WaitForChild("PlayerGui")

-- 1. TẠO SCREEN GUI (Màn hình chính chứa giao diện)
local ScreenGui = Instance.new("ScreenGui")
ScreenGui.Name = "KietHubGiaoDien"
ScreenGui.ResetOnSpawn = false
ScreenGui.Parent = playerGui

-- 2. TẠO MENU CHÍNH (Main Frame)
local MainFrame = Instance.new("Frame")
MainFrame.Name = "MainFrame"
MainFrame.Size = UDim2.new(0, 350, 0, 250) -- Rộng 350, Cao 250
MainFrame.Position = UDim2.new(0.5, -175, 0.5, -125) -- Căn giữa màn hình
MainFrame.BackgroundColor3 = Color3.fromRGB(25, 25, 25) -- Màu nền tối
MainFrame.BorderSizePixel = 0
MainFrame.Active = true
MainFrame.Draggable = true -- Cho phép kéo giữ di chuyển menu trên màn hình
MainFrame.Parent = ScreenGui

-- Tạo góc bo tròn cho Menu
local MainCorner = Instance.new("UICorner")
MainCorner.CornerRadius = UDim.new(0, 10)
MainCorner.Parent = MainFrame

-- 3. TIÊU ĐỀ MENU (Title)
local Title = Instance.new("TextLabel")
Title.Name = "Title"
Title.Size = UDim2.new(1, 0, 0, 40)
Title.BackgroundColor3 = Color3.fromRGB(35, 35, 35)
Title.Text = "KIET HUB - BETA v1.0"
Title.TextColor3 = Color3.fromRGB(255, 255, 255)
Title.TextSize = 18
Title.Font = Enum.Font.SourceSansBold
Title.Parent = MainFrame

local TitleCorner = Instance.new("UICorner")
TitleCorner.CornerRadius = UDim.new(0, 10)
TitleCorner.Parent = Title

-- 4. NÚT BẬT/TẮT CHỨC NĂNG (Toggle Button)
local ToggleButton = Instance.new("TextButton")
ToggleButton.Name = "ToggleFunction"
ToggleButton.Size = UDim2.new(0, 200, 0, 50)
ToggleButton.Position = UDim2.new(0.5, -100, 0.4, 0)
ToggleButton.BackgroundColor3 = Color3.fromRGB(200, 50, 50) -- Đỏ là đang tắt
ToggleButton.Text = "Auto Farm: OFF"
ToggleButton.TextColor3 = Color3.fromRGB(255, 255, 255)
ToggleButton.TextSize = 16
ToggleButton.Font = Enum.Font.SourceSans
ToggleButton.Parent = MainFrame

local ButtonCorner = Instance.new("UICorner")
ButtonCorner.CornerRadius = UDim.new(0, 8)
ButtonCorner.Parent = ToggleButton

-- Logic xử lý khi bấm nút Toggle
local isToggled = false
ToggleButton.MouseButton1Click:Connect(function()
    isToggled = not isToggled
    if isToggled then
        ToggleButton.BackgroundColor3 = Color3.fromRGB(50, 200, 50) -- Đổi sang xanh lá
        ToggleButton.Text = "Auto Farm: ON"
        print("Đã bật chức năng Auto!")
        -- Bạn có thể viết thêm logic xử lý hành động của game ở đây
    else
        ToggleButton.BackgroundColor3 = Color3.fromRGB(200, 50, 50) -- Đổi lại màu đỏ
        ToggleButton.Text = "Auto Farm: OFF"
        print("Đã tắt chức năng Auto!")
    end
end)

-- 5. NÚT ĐÓNG/MỞ NHANH MENU (Close/Open Button)
-- Nút này nằm góc màn hình để ẩn menu đi khi không dùng
local MenuButton = Instance.new("TextButton")
MenuButton.Name = "MenuButton"
MenuButton.Size = UDim2.new(0, 60, 0, 60)
MenuButton.Position = UDim2.new(0, 10, 0.5, -30) -- Nằm bên cạnh trái màn hình
MenuButton.BackgroundColor3 = Color3.fromRGB(0, 150, 255)
MenuButton.Text = "OPEN"
MenuButton.TextColor3 = Color3.fromRGB(255, 255, 255)
MenuButton.TextSize = 14
MenuButton.Font = Enum.Font.SourceSansBold
MenuButton.Parent = ScreenGui

local MenuButtonCorner = Instance.new("UICorner")
MenuButtonCorner.CornerRadius = UDim.new(1, 0) -- Biến nút thành hình tròn
MenuButtonCorner.Parent = MenuButton

-- Logic đóng mở Main Frame
MenuButton.MouseButton1Click:Connect(function()
    MainFrame.Visible = not MainFrame.Visible
    if MainFrame.Visible then
        MenuButton.Text = "CLOSE"
        MenuButton.BackgroundColor3 = Color3.fromRGB(150, 150, 150)
    else
        MenuButton.Text = "OPEN"
        MenuButton.BackgroundColor3 = Color3.fromRGB(0, 150, 255)
    end
end)
